<script type="text/babel">

const { useState, useEffect, useCallback } = React;

const API = "https://api.pokemontcg.io/v2";


const load = (k, fb) => { try { const v = localStorage.getItem(k); return v ? JSON.parse(v) : fb; } catch { return fb; } };

const save = (k, v) => { try { localStorage.setItem(k, JSON.stringify(v)); } catch {} };


const fmt = n => n != null ? `$${Number(n).toFixed(2)}` : null;

const getPrice = c =>

  c.tcgplayer?.prices?.holofoil?.market

  || c.tcgplayer?.prices?.reverseHolofoil?.market

  || c.tcgplayer?.prices?.normal?.market

  || c.tcgplayer?.prices?.["1stEditionHolofoil"]?.market

  || null;


function CardItem({ card, checked, onCheck, onRemove }) {

  const price = getPrice(card);

  return (

    <div className="card-item">

      {onCheck !== undefined && <input type="checkbox" checked={!!checked} onChange={e => onCheck(e.target.checked)} />}

      {card.images?.small && <img src={card.images.small} alt={card.name} loading="lazy" />}

      <div style={{flex:1, minWidth:0}}>

        <div className="card-name">{card.name}</div>

        <div className="card-sub">{card.set?.name} · {card.rarity || "Unknown"}</div>

        {card.number && <div className="card-num">#{card.number}/{card.set?.printedTotal || "?"}</div>}

      </div>

      <div className="card-price">

        {price ? fmt(price) : <span className="no-price">No price</span>}

        {onRemove && <button className="remove-btn" onClick={onRemove}>remove</button>}

      </div>

    </div>

  );

}


function CardSearch({ onAdd, placeholder }) {

  const [q, setQ] = useState("");

  const [results, setResults] = useState([]);

  const [loading, setLoading] = useState(false);

  const [searched, setSearched] = useState(false);

  const search = async () => {

    if (!q.trim()) return;

    setLoading(true); setSearched(true);

    try {

      const r = await fetch(`${API}/cards?q=name:"${encodeURIComponent(q.trim())}"&pageSize=20`);

      const d = await r.json();

      setResults(d.data || []);

    } catch { setResults([]); }

    setLoading(false);

  };

  return (

    <div>

      <div className="flex-gap" style={{marginBottom:12}}>

        <input type="text" value={q} onChange={e => setQ(e.target.value)} onKeyDown={e => e.key==="Enter" && search()} placeholder={placeholder || "Search cards..."} />

        <button onClick={search} style={{flexShrink:0}}>Search</button>

      </div>

      {loading && <div className="empty" style={{padding:"10px 0"}}>Searching...</div>}

      {searched && !loading && results.length === 0 && <div className="empty" style={{padding:"10px 0"}}>No results found.</div>}

      <div style={{maxHeight:340, overflowY:"auto"}}>

        {results.map(card => (

          <div key={card.id} className="card-item">

            {card.images?.small && <img src={card.images.small} alt={card.name} style={{width:36,height:50}} loading="lazy" />}

            <div style={{flex:1, minWidth:0}}>

              <div className="card-name" style={{fontSize:13}}>{card.name}</div>

              <div className="card-sub" style={{fontSize:11}}>{card.set?.name} · {card.rarity || "Unknown"}</div>

            </div>

            <button className="small-btn" onClick={() => onAdd(card)}>Add</button>

          </div>

        ))}

      </div>

    </div>

  );

}


function WishlistsTab({ wishlists, setWishlists }) {

  const [active, setActive] = useState(null);

  const [newName, setNewName] = useState("");

  const [showCreate, setShowCreate] = useState(false);

  const [showSearch, setShowSearch] = useState(false);

  const [renaming, setRenaming] = useState(null);

  const [renameVal, setRenameVal] = useState("");


  const create = () => {

    if (!newName.trim()) return;

    const id = Date.now().toString();

    const next = [...wishlists, { id, name: newName.trim(), cards: [] }];

    setWishlists(next); setNewName(""); setShowCreate(false); setActive(id);

  };

  const del = id => { setWishlists(wishlists.filter(l => l.id !== id)); if (active===id) setActive(null); };

  const addCard = (listId, card) => setWishlists(wishlists.map(l => l.id===listId ? {...l, cards:[...l.cards, card]} : l));

  const removeCard = (listId, cardId) => setWishlists(wishlists.map(l => l.id===listId ? {...l, cards: l.cards.filter(c=>c.id!==cardId)} : l));

  const rename = id => { if (!renameVal.trim()) return; setWishlists(wishlists.map(l => l.id===id ? {...l, name:renameVal.trim()} : l)); setRenaming(null); };


  const cur = wishlists.find(l => l.id === active);

  return (

    <div>

      <div className="pill-wrap">

        {wishlists.map(l => (

          <span key={l.id} className={`pill${active===l.id?" active":""}`} onClick={() => { setActive(l.id); setShowSearch(false); }}>

            {l.name} <span style={{fontSize:11, opacity:0.7, marginLeft:4}}>({l.cards.length})</span>

          </span>

        ))}

        <button onClick={() => setShowCreate(!showCreate)}>+ New List</button>

      </div>

      {showCreate && (

        <div className="flex-gap" style={{marginBottom:16}}>

          <input type="text" value={newName} onChange={e=>setNewName(e.target.value)} onKeyDown={e=>e.key==="Enter"&&create()} placeholder="List name..." />

          <button onClick={create}>Create</button>

          <button onClick={() => setShowCreate(false)}>Cancel</button>

        </div>

      )}

      {cur && (

        <div>

          <div className="row" style={{marginBottom:12}}>

            {renaming===cur.id ? (

              <>

                <input type="text" value={renameVal} onChange={e=>setRenameVal(e.target.value)} onKeyDown={e=>e.key==="Enter"&&rename(cur.id)} style={{maxWidth:200}} />

                <button onClick={()=>rename(cur.id)}>Save</button>

                <button onClick={()=>setRenaming(null)}>Cancel</button>

              </>

            ) : (

              <>

                <h3>{cur.name}</h3>

                <button className="small-btn" onClick={()=>{setRenaming(cur.id);setRenameVal(cur.name);}}>Rename</button>

                <button className="small-btn danger-btn" onClick={()=>del(cur.id)}>Delete list</button>

              </>

            )}

            <button onClick={()=>setShowSearch(!showSearch)} style={{marginLeft:"auto"}}>

              {showSearch ? "Hide search" : "+ Add cards"}

            </button>

          </div>

          {showSearch && <div className="search-box"><CardSearch onAdd={card=>addCard(cur.id,card)} placeholder="Search for a card..." /></div>}

          {cur.cards.length===0 && !showSearch && <div className="empty">No cards yet. Click "+ Add cards" to search.</div>}

          {cur.cards.map((card,i) => <CardItem key={card.id+i} card={card} onRemove={()=>removeCard(cur.id,card.id)} />)}

        </div>

      )}

      {!cur && wishlists.length>0 && <div className="empty">Select a wishlist above.</div>}

      {wishlists.length===0 && !showCreate && <div className="empty">No wishlists yet. Create one to get started.</div>}

    </div>

  );

}


function MasterSetsTab({ masterSets, setMasterSets }) {

  const [active, setActive] = useState(null);

  const [q, setQ] = useState("");

  const [setResults, setSetResults] = useState([]);

  const [loadingSet, setLoadingSet] = useState(false);

  const [loadingCards, setLoadingCards] = useState(false);

  const [showSearch, setShowSearch] = useState(false);


  const searchSets = async () => {

    if (!q.trim()) return;

    setLoadingSet(true);

    try {

      const r = await fetch(`${API}/sets?q=name:"${encodeURIComponent(q.trim())}"&pageSize=10`);

      const d = await r.json();

      setSetResults(d.data || []);

    } catch { setSetResults([]); }

    setLoadingSet(false);

  };


  const addSet = async (set) => {

    if (masterSets.find(s=>s.id===set.id)) { setShowSearch(false); setActive(set.id); return; }

    setLoadingCards(true);

    try {

      let all=[], page=1, total=999;

      while (all.length < total) {

        const r = await fetch(`${API}/cards?q=set.id:${set.id}&pageSize=250&page=${page}`);

        const d = await r.json();

        total = d.totalCount;

        all = [...all, ...(d.data||[])];

        if (!(d.data||[]).length) break;

        page++;

      }

      all.sort((a,b)=>(parseInt(a.number)||0)-(parseInt(b.number)||0));

      const next = [...masterSets, { id:set.id, name:set.name, logo:set.images?.logo, total:all.length, cards:all, checked:{} }];

      setMasterSets(next); setActive(set.id);

    } catch {}

    setLoadingCards(false); setShowSearch(false); setSetResults([]); setQ("");

  };


  const toggle = (setId, cardId, val) => setMasterSets(masterSets.map(s => s.id===setId ? {...s, checked:{...s.checked,[cardId]:val}} : s));

  const removeSet = id => { setMasterSets(masterSets.filter(s=>s.id!==id)); if(active===id) setActive(null); };


  const cur = masterSets.find(s=>s.id===active);

  return (

    <div>

      <div className="pill-wrap">

        {masterSets.map(s => {

          const owned = Object.values(s.checked).filter(Boolean).length;

          return (

            <span key={s.id} className={`pill${active===s.id?" active":""}`} onClick={()=>setActive(s.id)}>

              {s.name} <span style={{fontSize:11, opacity:0.7, marginLeft:4}}>({owned}/{s.total})</span>

            </span>

          );

        })}

        <button onClick={()=>setShowSearch(!showSearch)}>+ Add Set</button>

      </div>

      {showSearch && (

        <div className="search-box">

          <div className="flex-gap" style={{marginBottom:12}}>

            <input type="text" value={q} onChange={e=>setQ(e.target.value)} onKeyDown={e=>e.key==="Enter"&&searchSets()} placeholder="Search Pokémon sets..." />

            <button onClick={searchSets}>Search</button>

            <button onClick={()=>{setShowSearch(false);setSetResults([]);}}>Cancel</button>

          </div>

          {loadingSet && <div style={{fontSize:14,color:"var(--text2)"}}>Searching...</div>}

          {loadingCards && <div style={{fontSize:14,color:"var(--text2)"}}>Loading all cards... this may take a moment.</div>}

          {setResults.map(s => (

            <div key={s.id} className="card-item">

              {s.images?.logo && <img src={s.images.logo} alt={s.name} style={{height:28,width:"auto",objectFit:"contain",flexShrink:0}} />}

              <div style={{flex:1}}>

                <div className="card-name" style={{fontSize:13}}>{s.name}</div>

                <div className="card-sub" style={{fontSize:11}}>{s.series} · {s.printedTotal||s.total||"?"} cards</div>

              </div>

              <button className="small-btn" onClick={()=>addSet(s)}>Add</button>

            </div>

          ))}

        </div>

      )}

      {cur && (() => {

        const owned = Object.values(cur.checked).filter(Boolean).length;

        const pct = cur.total ? Math.round(owned/cur.total*100) : 0;

        return (

          <div>

            <div className="row" style={{marginBottom:12}}>

              {cur.logo && <img src={cur.logo} alt={cur.name} style={{height:32,objectFit:"contain"}} />}

              <div style={{flex:1}}>

                <div style={{fontWeight:500,fontSize:15}}>{cur.name}</div>

                <div style={{fontSize:12,color:"var(--text2)"}}>{owned} / {cur.total} cards ({pct}%)</div>

                <div className="progress-bar"><div className="progress-fill" style={{width:`${pct}%`}} /></div>

              </div>

              <button className="small-btn danger-btn" onClick={()=>removeSet(cur.id)}>Remove set</button>

            </div>

            <div className="scrollable">

              {cur.cards.map(card => <CardItem key={card.id} card={card} checked={!!cur.checked[card.id]} onCheck={v=>toggle(cur.id,card.id,v)} />)}

            </div>

          </div>

        );

      })()}

      {!cur && masterSets.length>0 && <div className="empty">Select a set above.</div>}

      {masterSets.length===0 && !showSearch && <div className="empty">No master sets yet. Click "+ Add Set" to begin.</div>}

    </div>

  );

}


function BuyListTab({ wishlists, masterSets, setMasterSets }) {

  const sections = [];

  masterSets.forEach(s => {

    const needed = s.cards.filter(c=>!s.checked[c.id]);

    if (needed.length) sections.push({ id:s.id, name:s.name, logo:s.logo, cards:needed, type:"master" });

  });

  wishlists.forEach(l => {

    if (l.cards.length) sections.push({ id:l.id, name:l.name, logo:null, cards:l.cards, type:"wishlist" });

  });

  const grand = sections.reduce((t,s)=>t+s.cards.reduce((st,c)=>st+(getPrice(c)||0),0),0);

  if (!sections.length) return <div className="empty">Your buy list is empty. Add master sets or wishlists to populate it.</div>;

  return (

    <div>

      <div className="grand-total">

        <div><div className="total-label">Estimated total</div><div className="total-val">{fmt(grand)}</div></div>

        <div style={{textAlign:"right"}}><div className="total-label">Cards needed</div><div className="total-val">{sections.reduce((t,s)=>t+s.cards.length,0)}</div></div>

      </div>

      {sections.map(sec => {

        const sub = sec.cards.reduce((t,c)=>t+(getPrice(c)||0),0);

        return (

          <div key={sec.id} style={{marginBottom:20}}>

            <div className="section-header">

              {sec.logo && <img src={sec.logo} alt={sec.name} style={{height:22,objectFit:"contain"}} />}

              {!sec.logo && <span className="wishlist-tag">wishlist</span>}

              <span className="section-name">{sec.name}</span>

              <span className="section-meta">{sec.cards.length} cards · {fmt(sub)}</span>

            </div>

            {sec.cards.map(card => (

              <CardItem key={card.id} card={card}

                checked={sec.type==="master"?false:undefined}

                onCheck={sec.type==="master"?(v)=>setMasterSets(masterSets.map(s=>s.id===sec.id?{...s,checked:{...s.checked,[card.id]:v}}:s)):undefined}

              />

            ))}

          </div>

        );

      })}

    </div>

  );

}


function StatsTab({ wishlists, masterSets }) {

  const totalOwned = masterSets.reduce((t,s)=>t+Object.values(s.checked).filter(Boolean).length,0);

  const totalNeeded = masterSets.reduce((t,s)=>t+s.cards.filter(c=>!s.checked[c.id]).length,0);

  const wTotal = wishlists.reduce((t,l)=>t+l.cards.length,0);

  const buyVal = masterSets.reduce((t,s)=>t+s.cards.filter(c=>!s.checked[c.id]).reduce((st,c)=>st+(getPrice(c)||0),0),0)

    + wishlists.reduce((t,l)=>t+l.cards.reduce((st,c)=>st+(getPrice(c)||0),0),0);

  return (

    <div>

      <div className="stat-grid">

        {[["Cards owned (tracked)",totalOwned],["Cards still needed",totalNeeded],["Wishlist cards",wTotal],["Est. buy list value",fmt(buyVal)||"$0.00"]].map(([l,v])=>(

          <div key={l} className="stat-card"><div className="stat-label">{l}</div><div className="stat-val">{v}</div></div>

        ))}

      </div>

      {masterSets.length>0 && (

        <div>

          <h3 style={{marginBottom:12}}>Master set progress</h3>

          {masterSets.map(s=>{

            const owned=Object.values(s.checked).filter(Boolean).length;

            const pct=s.total?Math.round(owned/s.total*100):0;

            return (

              <div key={s.id} style={{marginBottom:12}}>

                <div style={{display:"flex",justifyContent:"space-between",marginBottom:4}}>

                  <span style={{fontSize:13}}>{s.name}</span>

                  <span style={{fontSize:13,color:"var(--text2)"}}>{owned}/{s.total} ({pct}%)</span>

                </div>

                <div className="progress-bar"><div className="progress-fill" style={{width:`${pct}%`}} /></div>

              </div>

            );

          })}

        </div>

      )}

      {wishlists.length>0 && (

        <div style={{marginTop:20}}>

          <h3 style={{marginBottom:12}}>Wishlists</h3>

          {wishlists.map(l=>{

            const val=l.cards.reduce((t,c)=>t+(getPrice(c)||0),0);

            return (

              <div key={l.id} style={{display:"flex",justifyContent:"space-between",padding:"8px 12px",background:"var(--bg2)",borderRadius:8,marginBottom:6,border:"0.5px solid var(--border)"}}>

                <span style={{fontSize:13}}>{l.name}</span>

                <span style={{fontSize:13,color:"var(--text2)"}}>{l.cards.length} cards · {fmt(val)||"$0.00"}</span>

              </div>

            );

          })}

        </div>

      )}

      {!masterSets.length && !wishlists.length && <div className="empty">No data yet. Add master sets and wishlists to see stats.</div>}

    </div>

  );

}

            {t==="Master Sets"&&masterSets.length>0&&<span className="badge">{masterSets.length}</span>}

  );

}
